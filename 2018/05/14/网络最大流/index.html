#include <bits/stdc++.h>
#ifdef DEBUG
    #define MAXN 10
    #define MAXM 1000
#else
    #define MAXN 10000
    #define MAXM 100000
#endif
#define INF 2*MAXM+1
using namespace std;
struct Edge
{
    int to,next,cap,flow;
}e[2*MAXM+100];
int head[MAXN+1];
void addEdge(int from,int to,int cap)
{
    static int c=-1;
    e[++c]=(Edge){to,head[from],cap,0};
    head[from]=c;
    e[++c]=(Edge){from,head[to],0,0};
    head[to]=c;
}
bool vis[MAXN+1];
int dis[MAXN+1];
int nowhead[MAXN+1];
int n,m,s,t;
bool BFS()
{
    memset(vis,true,sizeof(vis));
    queue<int> q;
    q.push(s);
    dis[s]=0;
    vis[s]=false;
    while(!q.empty())
    {
        int x=q.front(); q.pop();
        for(int i=head[x]; i!=INF; i=e[i].next)
        {
            if (vis[e[i].to] && e[i].cap>e[i].flow)
            {
                vis[e[i].to]=false;
                dis[e[i].to]=dis[x]+1;
                q.push(e[i].to);
            }
        }
    }
    return !vis[t];
}
int DFS(int x,int a)
{
    if (x==t || a==0) return a;
    int flow=0,f;
    for(int &i=nowhead[x]; i!=INF; i=e[i].next)
    {
        if (dis[x]+1==dis[e[i].to] && (f=DFS(e[i].to,min(a,e[i].cap-e[i].flow)))>0)
        {
            e[i].flow+=f;
            e[i^1].flow-=f;
            flow+=f;
            a-=f;
            if (a==0) break;
        }
    }
    return flow;
}
int dinic()
{
    int flow=0;
    while(BFS())
    {
        copy(head,head+MAXN+1,nowhead);
        flow+=DFS(s,INT_MAX);
    }
    return flow;
}
int main()
{
    for(int i=1; i<=MAXN; i++) head[i]=INF;
    ios::sync_with_stdio(false);
    cin>>n>>m>>s>>t;
    for(int i=1; i<=m; i++)
    {
        int from,to,cap;
        cin>>from>>to>>cap;
        addEdge(from,to,cap);
    }
    cout<<dinic()<<'\n';
    return 0;
}
