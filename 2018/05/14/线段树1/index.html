using namespace std;

struct p
{
    long long left,right;
};p  tree[404040];
int ans;
long long sum[101010],add[101010];

int max(int x,int y){
    return x>y?x:y;
}

int min(int x,int y){
    return x<y?x:y;
}

int read(){
    int x=0,f=1;char ch=getchar();
    while (ch<'0' || ch>'9'){if (ch=='-')f=-1;ch=getchar();}
    while ('0'<=ch && ch<='9'){x=(x<<3)+(x<<1)+(ch^48);ch=getchar();}
    return x*f;
}

void write(long long x){
    int buf[50];
    if (x<0) putchar('-'),x=-x;
    buf[0]=0;
    while (x) buf[++buf[0]]=x%10,x/=10;
    if (!buf[0]) buf[0]=1,buf[1]=0;
    while (buf[0]) putchar('0'+buf[buf[0]--]);
}

void Build(int root,int l,int r)
{
    if (l==r)
    {
        int x=read();
        sum[root]=x;
        tree[root].left=l;tree[root].right=r;
        return ;
    }
    tree[root].left=l;tree[root].right=r;
    int mid=(l+r)>>1;
    Build(root<<1,l,mid);
    Build((root<<1)+1,mid+1,r);
    sum[root]+=sum[root<<1];
}

void Plus(int root,int l,int r,int x)
{
    int ll=tree[root].left,rr=tree[root].right;
    if (ll>=l&&rr<=r)
    {
        int t=x;
        sum[root]+=t*(rr-ll+1);
        add[root]+=x;
        return;
    }
    if (add[root])
    {
        add[root<<1]+=add[root];
        add[1+(root<<1)]+=add[root];
        int t=add[root];
        int len=rr-ll+1;
        sum[root<<1]+=t*((len+1)>>1);
        sum[(root<<1)+1]+=t*(len>>1);
        add[root]=0;
    }
    int mid=(ll+rr)>>1;
    if (mid>=l) Plus(root<<1,l,r,x);
    if (mid<r) Plus((root<<1)+1,l,r,x);
    sum[root]=sum[root<<1]+sum[(root<<1)+1];
}

int Ask(int root,int l,int r)
{
    int ll=tree[root].left;
    int rr=tree[root].right;
    if (ll>=l&&rr<=r) return sum[root];
    if (add[root])
    {
        add[root<<1]+=add[root];
        add[1+(root<<1)]+=add[root];
        int t=add[root];
        int len=rr-ll+1;
        sum[root<<1]+=t*((len+1)>>1);
        sum[(root<<1)+1]+=t*(len>>1);
        add[root]=0;
    }
    int mid=(ll+rr)>>1;
    int tmp=0;
    if (mid>=l) tmp+=Ask(root<<1,l,r);
    if (mid<r) tmp+=Ask((root<<1)+1,l,r);
    return tmp; 
}

int main(){
    int n=read(),m=read();
    Build(1,1,n);
    int q,w,e,q12;
    for (int i=1;i<=m;i++)
    {
        q=read();
        if (q==1)
        {
            q12=read();w=read();e=read();
            Plus(1,q12,w,e);
        }
        else 
        {
            w=read();e=read();
            write(Ask(1,w,e));
            putchar('\n');
        }
    }
    return 0;
}
