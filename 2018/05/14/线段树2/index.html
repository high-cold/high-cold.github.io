#include <iostream> //数据流输入/输出
#include <cstdio>
using namespace std;

int max(int x,int y){
    return x>y?x:y;
}

int min(int x,int y){
    return x<y?x:y;
}

int read(){
    int x=0,f=1;char ch=getchar();
    while (ch<'0'||ch>'9'){if (ch=='-')f=-1;ch=getchar();}
    while ('0'<=ch&&ch<='9'){x=x*10+(ch^48);ch=getchar();}
    return x*f;
}

inline void write(long long x){
    int buf[50];
    if (x<0) putchar('-'),x=-x;
    buf[0]=0;
    while (x) buf[++buf[0]]=x%10,x/=10;
    if (!buf[0]) buf[0]=1,buf[1]=0;
    while (buf[0]) putchar('0'+buf[buf[0]--]);
}

int p;

long long a[100007];

struct edge
{
    long long value,mult,add,l,r;
}tree[400007];

void Build(int root,int l,int r)
{
    tree[root].mult=1,
    tree[root].add=0,
    tree[root].l=l,
    tree[root].r=r;
    if(l==r) tree[root].value=a[l]%p;
    else
    {
        int m=(l+r)>>1;
        Build(root*2,l,m);
        Build(root*2+1,m+1,r);
        tree[root].value=(tree[root*2].value+tree[root*2+1].value)%p;
    }
    return ;
}

inline void pushdown(int root,int l,int r)
{
    int m=(l+r)>>1;
    tree[root*2].value=(tree[root*2].value*tree[root].mult+tree[root].add*(m-l+1))%p;
    tree[root*2+1].value=(tree[root*2+1].value*tree[root].mult+tree[root].add*(r-m))%p;
    tree[root*2].mult=(tree[root*2].mult*tree[root].mult)%p;
    tree[root*2+1].mult=(tree[root*2+1].mult*tree[root].mult)%p;
    tree[root*2].add=(tree[root*2].add*tree[root].mult+tree[root].add)%p;
    tree[root*2+1].add=(tree[root*2+1].add*tree[root].mult+tree[root].add)%p;
    tree[root].mult=1;
    tree[root].add=0;
    return ;
}

void Cheng(int root,int l,int r,long long k)
{
    if(r<tree[root].l||tree[root].r<l) return ;
    if(l<=tree[root].l&&tree[root].r<=r)
    {
        tree[root].value=(tree[root].value*k)%p;
        tree[root].mult=(tree[root].mult*k)%p;
        tree[root].add=(tree[root].add*k)%p;
        return ;
    }
    pushdown(root,tree[root].l,tree[root].r);
    Cheng(root*2,l,r,k);
    Cheng(root*2+1,l,r,k);
    tree[root].value=(tree[root*2].value+tree[root*2+1].value)%p;
    return ;
}

void Plus(int root,int l,int r,long long k)
{
    if(r<tree[root].l||tree[root].r<l) return ;
    if(l<=tree[root].l && tree[root].r<=r)
    {
        tree[root].add=(tree[root].add+k)%p;
        tree[root].value=(tree[root].value+k*(tree[root].r-tree[root].l+1))%p;
        return ;
    }
    pushdown(root,tree[root].l,tree[root].r);
    int m=(tree[root].l+tree[root].r)>>1;
    Plus(root*2,l,r,k);
    Plus(root*2+1,l,r,k);
    tree[root].value=(tree[root*2].value+tree[root*2+1].value)%p;
    return ;
}

long long Ask(int root,int l,int r)
{
    if(r<tree[root].l||tree[root].r<l) return 0;
    if(l<=tree[root].l&&tree[root].r<=r) return tree[root].value;
    pushdown(root,tree[root].l,tree[root].r);
    return (Ask(root*2,l,r)+Ask(root*2+1,l,r))%p;
}

int main(){
    int n,m;
    n=read(),m=read(),p=read(); 

    for(int i=1;i<=n;i++) a[i]=read();

    Build(1,1,n);
    int x,y,op;
    long long k;
    for (int i=1;i<=m;i++)
    {
        op=read();
        if(op==1)
        {
            x=read(),y=read(),k=read();
            Cheng(1,x,y,k);
            continue;
        }

        if(op==2)
        {
            x=read(),y=read(),k=read();
            Plus(1,x,y,k);
            continue;
        }

        if (op==3)
        {
            x=read(),y=read();
            write(Ask(1,x,y));
            putchar('\n');
        }
    }
    return 0;
}
