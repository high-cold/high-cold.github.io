Title:

给出一个由n个整数组成的序列。 我们称这个序列的一些连续的段为一个峡谷 ，当且仅当 al=ar a_l = a_r a​l​​=a​r​​ ，并且对于所有l<x<rl<x<rl<x<r，都有ax<ala_x<a_la​x​​<a​l​​成立。并定义一个峡谷的长度为 r−l r-l r−l 。 特别地，单独的一个数也是一个峡谷。 你的任务是回答qqq个询问：给定两个数l,rl,rl,r，求出的子段中最大长度的峡谷。

Input:

从文件 ravine.in 中读入数据。 输入第一行包含两个正整数n,qn,qn,q，表示序列的长度和询问数。 第二行包含nnn个正整数。 接下来的mmm行，每行包含两个数li,ril_i,r_il​i​​,r​i​​，表示一个询问。

Output

输出到文件 ravine.out 中。 对于每个询问，输出一个整数，表示所求的最大长度。

Example 1:

in

8 5 4 3 2 2 3 3 7 3 1 7 6 8 1 3 3 6 1 8

out

4 0 0 1 4

题解

用ST表维护区间最值（数据需要离散化）以及该值出现的最左和最右的位置stl,strstl,strstl,str，并计算每个点向右可以匹配的最长峡谷长度fff和向右可以匹配的最长峡谷长度ggg，然后依然用ST表维护f,gf,gf,g的区间最值stf,stgstf,stgstf,stg。 然后对于每个询问l,rl,rl,r，显然答案为max{str−stl,stf[l,stl−1],stg[str+1.r]}max\{str-stl,stf[l,stl-1],stg[str+1.r]\}max{str−stl,stf[l,stl−1],stg[str+1.r]}。

using namespace std;

int ans;

inline int max(int x,int y){return x>y?x:y;}

int read()
{
    int x=0,f=1;char ch=getchar();
    while (ch<'0' || ch>'9'){if (ch=='-')f=-1;ch=getchar();}
    while ('0'<=ch && ch<='9'){x=x*10+(ch^48);ch=getchar();}
    return x*f;
}

void write(int x)
{
    int buf[50];
    if (x<0) putchar('-'),x=-x;
    buf[0]=0;
    while (x) buf[++buf[0]]=x%10,x/=10;
    if (!buf[0]) buf[0]=1,buf[1]=0;
    while (buf[0]) putchar('0'+buf[buf[0]--]);
}

const int N=200200;

struct p
{
    int num,id;
} t[N];

int a[N],Next[N],stf[N][20],stg[N][20],f[N],g[N],mn[N],sl[N][20],sr[N][20],st[N][20],n;
vector <int> tong[N];
void prepare(int n)
{
    mn[0]=-1;
    for (int i=1;i<=n;i++)
    {
        mn[i]=((i&(i-1))==0)?mn[i-1]+1:mn[i-1];
        sl[i][0]=sr[i][0]=i;st[i][0]=a[i];
    }

    for (int j=1;j<=mn[n];j++)
        for (int i=1;i+(1<<j)-1<=n;i++)
        {
            if (st[i][j-1]<st[i+(1<<(j-1))][j-1])
            {
                sl[i][j]=sl[i+(1<<(j-1))][j-1];
                sr[i][j]=sr[i+(1<<(j-1))][j-1];
                st[i][j]=st[i+(1<<(j-1))][j-1];
            }
            else
            if (st[i][j-1]>st[i+(1<<(j-1))][j-1])
            {
                sl[i][j]=sl[i][j-1];
                sr[i][j]=sr[i][j-1];
                st[i][j]=st[i][j-1];
            }
            else
            {
                sl[i][j]=sl[i][j-1];
                sr[i][j]=sr[i+(1<<(j-1))][j-1];
                st[i][j]=st[i][j-1];
            }
        }
}

void F()
{
    for (int i=1;i<=n;i++) stf[i][0]=f[i];
    for (int j=1;j<=mn[n];j++)
        for (int i=1;i+(1<<j)-1<=n;i++) stf[i][j]=max(stf[i][j-1],stf[i+(1<<(j-1))][j-1]);
}

void G()
{
    for (int i=1;i<=n;i++) stg[i][0]=g[i];
    for (int j=1;j<=mn[n];j++)
        for (int i=1;i+(1<<j)-1<=n;i++) stg[i][j]=max(stg[i][j-1],stg[i+(1<<(j-1))][j-1]);
}

inline int query(int l,int r)
{
    if (l>r) return 0;
    int k=mn[r-l+1];
    return max(st[l][k],st[r-(1<<k)+1][k]);
}

inline int queryf(int l,int r)
{
    if (l>r) return 0;
    int k=mn[r-l+1];
    return max(stf[l][k],stf[r-(1<<k)+1][k]);
}

inline int queryg(int l,int r)
{
    if (l>r) return 0;
    int k=mn[r-l+1];
    return max(stg[l][k],stg[r-(1<<k)+1][k]);
}

bool cmp(const p x,const p y){return x.num<y.num||(x.num==y.num&&x.id<y.id);}

int main(){
    n=read();
    int T=read(),m,cnt=0;
    for (int i=1;i<=n;i++) t[i].num=a[i]=read(),t[i].id=i;
    sort(t+1,t+n+1,cmp);
    for (int i=1;i<=n;i++)
    {
        if (t[i].num!=t[i-1].num) cnt++;
        a[t[i].id]=cnt;
        tong[cnt].push_back(t[i].id);
    }
    prepare(n);
    for (int i=1,now=tong[i][tong[i].size()-1];i<=cnt;i++)
        for (int j=tong[i].size()-1;j>0;j--)
            if (query(tong[i][j-1],tong[i][j])>i) f[tong[i][j-1]]=0,now=tong[i][j-1];
            else f[tong[i][j-1]]=now; 

    for (int i=1,now=tong[i][0];i<=cnt;i++)
        for (int j=0;j<tong[i].size()-1;j++)
            if (query(tong[i][j],tong[i][j+1])>i) g[tong[i][j+1]]=0,now=tong[i][j+1];
            else g[tong[i][j+1]]=now;

    for (int i=1;i<=n;i++)
    {
        if (f[i]!=0) f[i]=f[i]-i;
        if (g[i]!=0) g[i]=i-g[i];
    }
    F();G();
    while (T--)
    {
        ans=0;
        int q=read(),w=read();
        int x=mn[w-q+1];
        if (st[q][x]>st[w-(1<<x)+1][x])
        {
            ans=max(queryf(q,sl[q][x]-1),queryg(sr[q][x]+1,w));
            ans=max(ans,sr[q][x]-sl[q][x]);
        }
        else
        if (st[q][x]<st[w-(1<<x)+1][x])
        {
            ans=max(queryf(q,sl[w-(1<<x)+1][x]-1),queryg(sr[w-(1<<x)+1][x]+1,w));
            ans=max(ans,sr[w-(1<<x)+1][x]-sl[w-(1<<x)+1][x]);
        }
        else
        {
            ans=max(queryf(q,sl[q][x]-1),queryg(sr[w-(1<<x)+1][x]+1,w));
            ans=max(ans,sr[w-(1<<x)+1][x]-sl[q][x]);
        }
        printf("%d\n",ans);
    }
    return 0;
}
